[docs]    def add_container(
        self,
        query_selector: str,
        width: str = 'auto',
        height: str = 'auto',
        display: str = 'flex',
        xOffset: str = '0',
        yOffset: str = '0',
        position: str = '',
        orientation: str = 'column',
        id: str = None,
        label: str = None
    ):
        if id is None:
            id = ''.join(random.choices(string.ascii_letters, k=10))
        if label is None:
            label = id

        self._steps.append(
            ContainerRendering(
                label=label,
                id=id,
                query_selector=query_selector,
                width=width,
                height=height,
                display=display,
                xOffset=xOffset,
                yOffset=yOffset,
                position=position,
                orientation=orientation
            )
        )

        return id

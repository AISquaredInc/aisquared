430    def get_model_filenames(self):
431        """
432        Get filenames for all models in the configuration
433        """
434        filenames = []
435        if isinstance(self.harvesting_steps[0], list):
436            harvesting_list = [
437                h for harvester in self.harvesting_steps for h in harvester]
438        else:
439            harvesting_list = self.harvesting_steps
440        for harvester in harvesting_list:
441            if isinstance(harvester, ModelConfiguration):
442                filenames.extend(harvester.get_model_filenames())
443
444        if isinstance(self.analytic[0], ANALYTIC_CLASSES):
445            for a in self.analytic:
446                if isinstance(a, LOCAL_CLASSES):
447                    filenames.append(a.path)
448        else:
449            for analytic in self.analytic:
450                for a in analytic:
451                    if isinstance(a, LOCAL_CLASSES):
452                        filenames.append(a.path)
453        return filenames

481    def compile(self, filename=None, dtype=None):
482        """
483        Compile the object into a '.air' file, which can then be dragged and
484        dropped into applications using the AI Squared JavaScript SDK
485
486        Parameters
487        ----------
488        filename : path-like or None (default None)
489            Filename to compile to. If None, defaults to '{NAME}.air', where {NAME} is the
490            name of the analytic
491        dtype : str or None (default None)
492            The datatype to use for the model weights. If None, defaults to 'float32'
493        """
494        if filename is None:
495            filename = self.name + '.air'
496
497        if dtype is None:
498            dtype_map = None
499        else:
500            dtype_map = {dtype: '*'}
501
502        dirname = os.path.join('.', os.path.splitext(filename)[0])
503
504        # write the object as json config
505        os.makedirs(dirname, exist_ok=True)
506        with open(os.path.join(dirname, 'config.json'), 'w') as f:
507            json.dump(self.to_dict(), f)
508
509        # go through the files and copy them/make them tfjs files
510        filenames = self.get_model_filenames()
511        if len(filenames) != 0:
512            for f in filenames:
513                if os.path.splitext(f)[-1] == '.h5':
514                    model = tf.keras.models.load_model(f)
515                    model_dir = os.path.join(dirname, os.path.split(f)[-1])
516                    tfjs.converters.save_keras_model(
517                        model, model_dir, quantization_dtype_map=dtype_map)
518                else:
519                    shutil.copy(f, dirname)
520
521        # go through the entire directory of dirname, grab all files, and make
522        # the archive file
523        shutil.make_archive(filename, 'zip', dirname)
524
525        # Move the archive file to just have .air
526        shutil.move(filename + '.zip', filename)
527
528        # Remove the temp directory
529        shutil.rmtree(dirname, ignore_errors=True)

504

507        with open(os.path.join(dirname, 'config.json'), 'w') as f:

515                    model = tf.keras.models.load_model(f)
516                    model_dir = os.path.join(dirname, os.path.split(f)[-1])
517                    tfjs.converters.save_keras_model(
518                        model, model_dir, quantization_dtype_map=dtype_map)
519                else:

521
522        # go through the entire directory of dirname, grab all files, and make

526        # Move the archive file to just have .air
527        shutil.move(filename + '.zip', filename)
528
529        # Remove the temp directory
530        shutil.rmtree(dirname, ignore_errors=True)
